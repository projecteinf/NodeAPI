```javascript
// El Formatador de Durada
const formatar2digits = (num) => {
	if (num < 10) {
		num = "0" + num
	}
	return num
}

const formatarTemps = (duradaSegons) => {
	const minuts = Math.floor(duradaSegons / 60)
	const segons = duradaSegons % 60
	
	return formatar2digits(minuts) +":"+ formatar2digits(segons)
}

console.log(formatarTemps(32))
console.log(formatarTemps(65))
console.log(formatarTemps(115))
```
