# El Formulari Reactiu (Interceptar el *Submit*)
```html
<!DOCTYPE html>
<html lang="ca">
    <head>
        <meta charset="UTF-8">
        <title>Spotify de Classe - DOM</title>
        <style>
            body { background-color: #121212; color: white; font-family: sans-serif; padding: 20px; }

            .formulari-afegir { background-color: #181818; padding: 15px; border-radius: 8px; margin-bottom: 20px; border: 1px solid #282828; }
            .formulari-afegir input { background: #282828; border: 1px solid #3e3e3e; color: white; padding: 8px; margin-right: 10px; border-radius: 4px; }
            .formulari-afegir button { background-color: #1DB954; border: none; color: white; padding: 8px 15px; border-radius: 20px; font-weight: bold; cursor: pointer; }
            
            .barra-reproductor { position: fixed; bottom: 0; left: 0; right: 0; background-color: #040404; border-top: 1px solid #282828; padding: 20px; display: flex; justify-content: space-between; align-items: center; }
            .info-reproduccio { color: #1DB954; font-weight: bold; }

            .playlist-title { color: #1DB954; } /* Verd Spotify */
            .grid-cancons { display: grid; grid-template-columns: repeat(auto-fill, minmax(200px, 1fr)); gap: 15px; margin-top: 20px; }
            .targeta-canco { background-color: #181818; padding: 15px; border-radius: 8px; border: 1px solid #282828; }
            .targeta-canco h1 { margin: 0 0 10px 0; font-size: 16px; }
            .targeta-canco p { margin: 5px 0; color: #b3b3b3; font-size: 14px; }
        </style>
    </head>
    <body>
        <form id="formulari-afegir-canco" class="formulari-afegir">
            <h3>Afegir Nova Cançó</h3>
            <input type="text" id="input-titol" placeholder="Títol de la cançó" required>
            <input type="text" id="input-artista" placeholder="Artista" required>
            <input type="text" id="input-genere" placeholder="Gènere" required>
            <button type="submit">+ Afegir al catàleg</button>
        </form>

        <h1 class="playlist-title">El teu Catàleg Musical</h1>
        <div id="contenedor-principal" class="grid-cancons"></div>

        <div class="barra-reproductor">
            <div>Ara mateix sona: <span id="reproductor-text" class="info-reproduccio">Cap cançó seleccionada</span></div>
            <div style="color: #b3b3b3;"></div>
        </div>

        <script>
            const baseDeDadesMusica = [
                { id: 1, titol: "Despechá", artista: "Rosalía", durada: 157, genere: "Pop" },
                { id: 2, titol: "Candy", artista: "Rosalía", durada: 191, genere: "Pop" },
                { id: 3, titol: "Monaco", artista: "Bad Bunny", durada: 207, genere: "Trap" },
                { id: 4, titol: "Ojitos Lindos", artista: "Bad Bunny", durada: 265, genere: "Reggaeton" }
            ];

            const formCanco = document.querySelector('#formulari-afegir-canco');
            formCanco.addEventListener("submit", (e) => {
                e.preventDefault();
                canco = {
                    titol: formCanco.querySelector("#input-titol").value,
                    artista: formCanco.querySelector("#input-artista").value,
                    genere: formCanco.querySelector("#input-genere").value
                };
                baseDeDadesMusica.push(canco);
                formCanco.reset();
                renderitzarCataleg(baseDeDadesMusica,document.querySelector("#contenedor-principal"));
            });

            const reproduirCanco = (canco) => {
                const elReproductor = document.querySelector(".barra-reproductor div");
                elReproductor.textContent = canco.titol + " - " + canco.artista;
            }

            const crearTargetaHtml = (canco) => {
                const divCanco = document.createElement("div");
                divCanco.classList.add("targeta-canco");

                const titol = document.createElement("h1");
                titol.textContent = canco.titol;

                const artista = document.createElement("p");
                artista.textContent = canco.artista;
                
                
                divCanco.appendChild(titol);
                divCanco.appendChild(artista);
                divCanco.addEventListener('click', (e) => {
                    reproduirCanco(canco);
                });
                return divCanco;
            }
            const renderitzarCataleg = (cancons,contenidor) => {
                contenidor.textContent="";
                cancons.forEach(canco => {
                    contenidor.appendChild(crearTargetaHtml(canco))
                });
            }
         renderitzarCataleg(baseDeDadesMusica,document.querySelector("#contenedor-principal"));

        </script>
    </body>
</html>
```
