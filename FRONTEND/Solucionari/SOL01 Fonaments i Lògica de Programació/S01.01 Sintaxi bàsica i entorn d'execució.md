```javascript
const USUARI_CORRECTE = "profe_web";
const CONTRASENYA_CORRECTE = "12345";
const EDAT_MINIMA = 14;

let usuariIntent = "profe_web";
let passIntent = "12345";
let edatIntent = Number("16"); // Conversió explícita de tipus sol·licitada

debugger;

if (usuariIntent === USUARI_CORRECTE && passIntent === CONTRASENYA_CORRECTE) {
    
    // Si les credencials són bones, comprovem l'edat
    if (edatIntent >= EDAT_MINIMA) {
        console.log("Benvingut a Spotify de classe. Carregant la teva música...");
        for (let i = 1; i <= 3; i++) {
            console.log("Configurant gènere musical favorit número " + i + "...");
        }
        
    } else {
        console.log("Error: Ets massa jove per utilitzar aquesta aplicació.");
    }

} else {
    console.log("Credencials incorrectes. Accés denegat.");
}
```
